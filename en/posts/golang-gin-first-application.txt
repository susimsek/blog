```go

	"github.com/gin-gonic/gin"
)

func main() {
	r := gin.Default()

	r.GET("/hello", func(c *gin.Context) {
		c.JSON(200, gin.H{

		})
	})

	r.Run() // Listen and serve on 0.0.0.0:8080
}
```
